%%%% QUESTĂO 01 %%%%
\questao{
	Seja $M^{n}$ uma $n$-variedade topolĂłgica. Mostre que:
	\begin{enumerate}[label=(\alph*)]
		\item Qualquer atlas suave para $M^{n}$ estĂĄ contido em um atlas suave maximal (Ășnico).
		\item Dois atlas suaves para $M^{n}$ determinam o mesmo atlas suave maximal se, e somente se, a uniĂŁo desses atlas Ă© um atlas suave.
	\end{enumerate}
}
\solucao{
	
	\textbf{AnĂĄlise do Enunciado}
	\begin{itemize}
		\item \textbf{HipĂłteses:} $M^n$ Ă© uma $n$-variedade topolĂłgica.
		\item \textbf{Tese (a):} Provar a existĂȘncia e unicidade de um atlas suave maximal que contenha qualquer atlas suave dado.
		\item \textbf{Tese (b):} Caracterizar a equivalĂȘncia de atlas suaves atravĂ©s da suavidade de sua uniĂŁo.
	\end{itemize}
	\vspace{5pt}
	\textbf{(a)} Seja $\mathcal{A} = \{(U_\alpha, \varphi_\alpha)\}_{\alpha \in \Lambda}$ um atlas suave para $M^n$. Definimos a coleĂ§ĂŁo $\overline{\mathcal{A}}$ como o conjunto de todas as cartas coordenadas que sĂŁo suavemente compatĂ­veis com todas as cartas de $\mathcal{A}$:
	\[ \overline{\mathcal{A}} = \{ (V, \psi) : (V, \psi) \text{ Ă© suavemente compatĂ­vel com toda carta } (U_\alpha, \varphi_\alpha) \in \mathcal{A} \}. \]
	
	Claramente $\mathcal{A} \subseteq \overline{\mathcal{A}}$, pois as cartas de um atlas suave sĂŁo, por definiĂ§ĂŁo, compatĂ­veis entre si. Para mostrar que $\overline{\mathcal{A}}$ Ă© um atlas suave, devemos provar que quaisquer duas cartas $(V_1, \psi_1), (V_2, \psi_2) \in \overline{\mathcal{A}}$ sĂŁo suavemente compatĂ­veis. 
	
	Seja $p \in V_1 \cap V_2$. Como $\mathcal{A}$ Ă© um atlas, existe uma carta $(U_\alpha, \varphi_\alpha) \in \mathcal{A}$ tal que $p \in U_\alpha$. Definimos o aberto $W = V_1 \cap V_2 \cap U_\alpha$. A aplicaĂ§ĂŁo de transiĂ§ĂŁo $\psi_2 \circ \psi_1^{-1}$ restrita ao domĂ­nio $\psi_1(W) \subset \mathbb{R}^n$ pode ser fatorada como:
	\[ \psi_2 \circ \psi_1^{-1} \Big|_{\psi_1(W)} = \underbrace{(\psi_2 \circ \varphi_\alpha^{-1})}_{\text{suave}} \circ \underbrace{(\varphi_\alpha \circ \psi_1^{-1})}_{\text{suave}}. \]
	
	Visto que ambas as funĂ§Ă”es no lado direito sĂŁo suaves (pela definiĂ§ĂŁo de $\overline{\mathcal{A}}$), a sua composiĂ§ĂŁo tambĂ©m Ă© suave em uma vizinhanĂ§a de $\psi_1(p)$. Como isso Ă© vĂĄlido para todo ponto na interseĂ§ĂŁo, concluĂ­mos que $\overline{\mathcal{A}}$ Ă© um atlas suave.
	
	\textbf{Maximalidade:} Para mostrar que $\overline{\mathcal{A}}$ Ă© maximal, seja $(V, \psi)$ uma carta suave compatĂ­vel com todas as cartas de $\overline{\mathcal{A}}$. Como $\mathcal{A} \subseteq \overline{\mathcal{A}}$, em particular, $(V, \psi)$ deve ser compatĂ­vel com todas as cartas de $\mathcal{A}$. Por definiĂ§ĂŁo de $\overline{\mathcal{A}}$, qualquer carta compatĂ­vel com $\mathcal{A}$ jĂĄ pertence a $\overline{\mathcal{A}}$. Logo, $(V, \psi) \in \overline{\mathcal{A}}$, o que prova que $\overline{\mathcal{A}}$ Ă© maximal.
	
	\textbf{Unicidade:} Suponha que exista outro atlas maximal $\mathcal{C}$ tal que $\mathcal{A} \subseteq \mathcal{C}$. Como $\mathcal{C}$ Ă© um atlas suave, todas as suas cartas sĂŁo mutuamente compatĂ­veis. Como $\mathcal{A} \subseteq \mathcal{C}$, todas as cartas de $\mathcal{C}$ sĂŁo compatĂ­veis com todas as cartas de $\mathcal{A}$. Pela definiĂ§ĂŁo de $\overline{\mathcal{A}}$, isto implica $\mathcal{C} \subseteq \overline{\mathcal{A}}$. Por maximalidade de $\mathcal{C}$, temos $\mathcal{C} = \overline{\mathcal{A}}$.
	
	\textbf{(b)} $(\implies)$ Suponha que $\mathcal{A}$ e $\mathcal{B}$ determinam o mesmo atlas suave maximal $\mathcal{M}$. Por definiĂ§ĂŁo, $\mathcal{A} \subseteq \mathcal{M}$ e $\mathcal{B} \subseteq \mathcal{M}$. Logo, $\mathcal{A} \cup \mathcal{B} \subseteq \mathcal{M}$. Como subconjuntos de cartas de um atlas suave sĂŁo mutuamente compatĂ­veis, $\mathcal{A} \cup \mathcal{B}$ Ă© um atlas suave.
	
	$(\impliedby)$ Suponha agora que $\mathcal{A} \cup \mathcal{B}$ seja um atlas suave. Seja $\mathcal{M}$ o atlas suave maximal Ășnico que contĂ©m a uniĂŁo $\mathcal{A} \cup \mathcal{B}$ (cuja existĂȘncia foi garantida no item (a)). 
	
	Como $\mathcal{A} \subseteq (\mathcal{A} \cup \mathcal{B}) \subseteq \mathcal{M}$, entĂŁo $\mathcal{M}$ Ă© um atlas suave maximal que contĂ©m $\mathcal{A}$. Pela unicidade, $\mathcal{A}_{max} = \mathcal{M}$. Da mesma forma, como $\mathcal{B} \subseteq (\mathcal{A} \cup \mathcal{B}) \subseteq \mathcal{M}$, temos $\mathcal{B}_{max} = \mathcal{M}$. Portanto, $\mathcal{A}_{max} = \mathcal{B}_{max}$.
}
%%%% QUESTĂO 02 %%%%
\questao{
	Mostre que uma $n$-variedade suave $M^{n}$ Ă© conexa se, e somente se, ela Ă© conexa por caminhos.
}
\solucao{
	
	\textbf{AnĂĄlise do Enunciado}
	\begin{itemize}
		\item \textbf{HipĂłteses:} $M^n$ Ă© uma $n$-variedade suave.
		\item \textbf{Tese:} Provar a equivalĂȘncia entre a conexidade topolĂłgica e a conexidade por caminhos.
	\end{itemize}
	\vspace{5pt}
	\textbf{($\Leftarrow$)} Suponha que $M$ seja conexa por caminhos. Provaremos que $M$ Ă© conexa por contradiĂ§ĂŁo. Se $M$ fosse desconexa, existiriam abertos $U, V \subset M$ nĂŁo vazios e disjuntos tais que $M = U \cup V$. Sejam $p \in U$ e $q \in V$. Por hipĂłtese, existe $\gamma: [0,1] \to M$ contĂ­nua com $\gamma(0) = p$ e $\gamma(1) = q$.
	
	Os conjuntos $A = \gamma^{-1}(U)$ e $B = \gamma^{-1}(V)$ seriam abertos em $[0,1]$ pela continuidade de $\gamma$. TerĂ­amos $A \cup B = [0,1]$, $A \cap B = \emptyset$, com $0 \in A$ e $1 \in B$ (logo ambos nĂŁo vazios). Isso contradiz a conexidade de $[0,1]$. Portanto, $M$ Ă© conexa.
	
	\textbf{($\Rightarrow$)} Suponha $M$ conexa. Fixemos $p \in M$ e definamos o conjunto de pontos que podem ser ligados a $p$ por um caminho contĂ­nuo:
	\[ A = \{ q \in M : \exists \, \sigma: [0,1] \to M \text{ contĂ­nua com } \sigma(0)=p \text{ e } \sigma(1)=q \}. \]
	
	\textbf{1. $A$ Ă© aberto:} Seja $q \in A$. Existe uma vizinhanĂ§a coordenada $U$ de $q$ homeomorfa a uma bola em $\mathbb{R}^n$, a qual Ă© conexa por caminhos. Seja $\phi: U \to B^n$ o homeomorfismo. Para qualquer $q' \in U$, definimos o caminho interno $\alpha: [0,1] \to U$ por $\alpha(t) = \phi^{-1}\left((1-t)\phi(q) + t\phi(q')\right)$. Como $q \in A$, existe $\sigma: [0,1] \to M$ de $p$ a $q$. Definimos a concatenaĂ§ĂŁo $\beta: [0,1] \to M$:
	\[ \beta(t) = \begin{cases} \sigma(2t), & 0 \leq t \leq 1/2 \\ \alpha(2t-1), & 1/2 < t \leq 1 \end{cases} \]
	$\beta$ Ă© contĂ­nua pelo Lema da Colagem ($\sigma(1)=q=\alpha(0)$) e liga $p$ a $q'$. Logo $U \subseteq A$, e $A$ Ă© aberto.
	
	\textbf{2. $A$ Ă© fechado:} Seja $r \in \overline{A}$. Seja $V$ uma vizinhanĂ§a coordenada de $r$ (homeomorfa a uma bola). Pela definiĂ§ĂŁo de fecho, existe $s \in V \cap A$. Como $s \in A$, ele se liga a $p$. Como $s$ e $r$ estĂŁo na mesma vizinhanĂ§a conexa por caminhos $V$, eles se ligam. Por concatenaĂ§ĂŁo, $r \in A$. Logo $\overline{A} = A$.
	
	Como $M$ Ă© conexo e $A$ Ă© aberto, fechado e nĂŁo vazio, $A = M$.
}
%%%% QUESTĂO 03 %%%%
\questao{
	Seja $\mathbb{S}^{n}=\{(x^{1},...,x^{n+1})\in\mathbb{R}^{n+1}; (x^{1})^{2}+\dots+(x^{n+1})^{2}=1\}$ a esfera unitĂĄria. Prove que:
	\begin{enumerate}[label=(\alph*)]
		\item $\mathbb{S}^{n}$ Ă© uma $n$-variedade suave, compacta e de Hausdorff com base enumerĂĄvel exibindo um atlas suave sobre $\mathbb{S}^{n}$.
		\item $\mathbb{S}^{n}$ Ă© a imagem inversa de um valor regular de uma funĂ§ĂŁo apropriada.
		\item Mostre que existe um atlas suave de $\mathbb{S}^{n}$ com apenas duas cartas.
		\item $\hat{E}$ possĂ­vel obtermos um atlas suave para $\mathbb{S}^{n}$ com apenas uma carta?
	\end{enumerate}
}
\solucao{
	
	\textbf{AnĂĄlise do Enunciado}
	\begin{itemize}
		\item \textbf{HipĂłteses:} $\mathbb{S}^n$ definida como a esfera unitĂĄria em $\mathbb{R}^{n+1}$.
		\item \textbf{Tese (a):} Provar que $\mathbb{S}^n$ Ă© uma $n$-variedade suave, compacta, Hausdorff e com base enumerĂĄvel.
		\item \textbf{Tese (b):} Representar $\mathbb{S}^n$ como $f^{-1}(c)$ com $c$ valor regular.
		\item \textbf{Tese (c):} Construir um atlas de duas cartas.
		\item \textbf{Tese (d):} Provar a impossibilidade de um atlas de uma Ășnica carta.
	\end{itemize}
	\vspace{5pt}
	\textbf{(a)} Como $\mathbb{S}^n \subset \mathbb{R}^{n+1}$, entĂŁo pela topologia induzida ela herda as propriedades de \textbf{Hausdorff} e \textbf{base enumerĂĄvel}. 
	
	\textbf{1. Compacidade}
	
	Seja $f: \mathbb{R}^{n+1} \to \mathbb{R}$ dada por $f(x) = |x|^2$. Como $f$ Ă© contĂ­nua e $\mathbb{S}^n = f^{-1}(1)$, a esfera Ă© um conjunto fechado. Sendo tambĂ©m limitada ($|x|=1$), pelo Teorema de Heine-Borel, $\mathbb{S}^n$ Ă© compacta.
	
	\textbf{2. ConstruĂ§ĂŁo do Atlas das Calotas Polares}
	
	Para cada $i\in\{1,\ldots,n+1\}$, definimos os abertos $U_i^{\pm} = \{x \in \mathbb{S}^n : \pm x^i > 0\}$ e 
	\begin{eqnarray*}
		\varphi_i^{\pm}: U_i^{\pm}&\longrightarrow &\mathbb{R}^n\\
		(x^1, \dots,x^{n+1})&\longmapsto& (x^1, \dots, \widehat{x^i}, \dots, x^{n+1})
	\end{eqnarray*}
	onde, $\widehat{x^i}$ denota que $x_i$ Ă© omitido.
	
	\textbf{Prova de Cobertura}: Seja $x = (x^1, \dots, x^{n+1}) \in \mathbb{S}^n$. Como $|x|^2 = 1$, Ă© impossĂ­vel que todas as coordenadas de $x$ sejam nulas. Portanto, existe pelo menos um $i \in \{1, \dots, n+1\}$ tal que $x^i \neq 0$. Se $x^i > 0$, entĂŁo $x \in U_i^+$; se $x^i < 0$, entĂŁo $x \in U_i^-$.	Isso implica que $\mathbb{S}^n = \bigcup_{i=1}^{n+1} (U_i^+ \cup U_i^-)$.
	
	Notemos que $\varphi_i^{\pm}$ sĂŁo contĂ­nuas e alĂ©m disso, $\varphi_i^{\pm}( U_i^{\pm})=\mathbb{B}^n$, onde $\mathbb{B}^n=\{x\in\mathbb{R}^n;|x|<1\}$. De fato, dado $x\in U_i^{\pm}$, temos que:
	$$|\varphi_i^{\pm}(x)|^2=(x^1)^2+\cdots+(x^{i-1})^2+(x^{i+1})^2+\cdots+(x^n)^2<(x^1)^2+\cdots+(x^{i-1})^2+(x^i)^2+(x^{i+1})^2+\cdots+(x^n)^2=1.$$ 
	
	\textbf{CĂĄlculo da Inversa}: A inversa $(\varphi_i^{\pm})^{-1}: \mathbb{B}^n\rightarrow U_i^{\pm}$, Ă© obtida isolando $x^i$:
	$$(x^i)^2 = 1 - \sum_{k \neq i} (x^k)^2 \implies x^i = \pm \sqrt{1 - |u|^2},$$ 
	donde
	$$(\varphi_i^{\pm})^{-1}(u)=(u^1, \dots, u^{i-1},\pm\sqrt{1-|u|^2}, u^{i+1}, \dots, u^n),$$ que tambĂ©m sĂŁo contĂ­nuas.
	
	AlĂ©m disso, temos que $(\varphi_i^{\pm})^{-1}(\mathbb{B}^n)=U_i^{\pm}$, pois
	$$|(\varphi_i^{\pm})^{-1}(u)|^2=(u^1)^2+\cdots+(u^{i-1})^2+1-|u|^2+(u^{i+1})^2+\cdots(u^n)^2=1.$$
	
	Assim, cada uma das $2(n+1)$ cartas $\varphi_i^{\pm}$ Ă© um homeomorfismo de $U_i^{\pm}$ sobre o aberto $\mathbb{B}^n \subset \mathbb{R}^n$, donde $\{(U_i^{\pm}, \varphi_i^{\pm})\}$ constitui um atlas topolĂłgico para $\mathbb{S}^n$. Portanto, $\mathbb{S}^n$ Ă© uma $n$-variedade topolĂłgica.
	
	\textbf{3. Suavidade das TransiĂ§Ă”es}
	
	A transiĂ§ĂŁo $\psi = \varphi_j^{\pm} \circ (\varphi_i^{\pm})^{-1}$ (para $i < j$) em $u \in \varphi_i^+(U_i^+ \cap U_j^+)\subset\mathbb{B}^n$ resulta em:
	$$ \psi(u) = (u^1, \dots, \pm\sqrt{1-|u|^2}, \dots, \widehat{u^j}, \dots, u^n) $$
	Como $|u|^2 < 1$, entĂŁo a raiz quadrada Ă© suave em $(0, \infty)$, e portanto a transiĂ§ĂŁo Ă© suave. De forma similar, vale quando $i>j$. Quando $i=j$, entĂŁo $\varphi_j^{\pm}\circ(\varphi_i^{\pm})^{-1}=\operatorname{Id}_{\mathbb{B}^n}$ que Ă© suave.
	
	ConcluĂ­mos que, $\{U_i^{\pm},\varphi_i^{\pm}\}$ Ă© um atlas suave para $\mathbb{S}^n$, e portanto a esfera Ă© uma $n$-variedade suave com essa estrutura.
	
	\textbf{(b)} Consideremos a funĂ§ĂŁo diferenciĂĄvel,
	\begin{eqnarray*}
		f: \mathbb{R}^{n+1}&\longrightarrow &\mathbb{R}\\
		x&\longmapsto& f(x)=\langle x,x\rangle=|x|^2.
	\end{eqnarray*}
	Temos que $\nabla f(x) = 2x$ que Ă© nĂŁo nulo para todo $x\neq \vec{0}$. Em particular, $1$ Ă© valor regular de $f$, donde $\mathbb{S}^n=f^{-1}(1)$.
	
	\textbf{(c)} Exibiremos o atlas de duas cartas via ProjeĂ§ĂŁo EstereogrĂĄfica. Sejam $U_N = \mathbb{S}^n \setminus \{N\}$ e $U_S = \mathbb{S}^n \setminus \{S\}$, onde $N = (0,\dots,1)$ e $S = (0,\dots,-1)$. Notemos que, $\mathbb{S}^n=U_N\cup U_S$, ou seja, temos uma cobertura.

	\textbf{CĂĄlculo da Inversa $\phi_N^{-1}$}: Seja $u \in \mathbb{R}^n$, queremos $x \in U_N$ tal que $u^i = \frac{x^i}{1-x^{n+1}}$, ou seja, 
	$$\boxed{x^i = u^i(1-x^{n+1}).}\quad (\ast)$$ 
	Elevando ao quadrado e somando de $i=1$ atĂ© $n$:
	$$\sum_{i=1}^n (x^i)^2 = \sum_{i=1}^n [u^i(1-x^{n+1})]^2 = (1-x^{n+1})^2 \sum_{i=1}^n (u^i)^2 = |u|^2 (1-x^{n+1})^2$$
	Como $x \in U_N$, temos $\sum_{i=1}^n (x^i)^2 = 1 - (x^{n+1})^2$. Igualando as expressĂ”es:
	\begin{align*}
		1 - (x^{n+1})^2 &= |u|^2 (1-x^{n+1})^2 \\
		(1-x^{n+1})(1+x^{n+1}) &= |u|^2 (1-x^{n+1})^2 \\
		1+x^{n+1} &= |u|^2 (1-x^{n+1}) \quad (\text{pois } x \in U_N \implies x^{n+1} \neq 1) \\
		x^{n+1}(1+|u|^2) &= |u|^2 - 1 \implies x^{n+1} = \frac{|u|^2-1}{|u|^2+1}
	\end{align*}
	Substituindo $x^{n+1}$  em $(\ast)$, obtemos que: $$x^i = u^i \left(1 - \frac{|u|^2-1}{|u|^2+1}\right) = u^i \left(\frac{|u|^2+1 - |u|^2 + 1}{|u|^2+1}\right) = \frac{2u^i}{|u|^2+1}.$$
	
	Logo, $\phi_N^{-1}:\mathbb{R}^n\setminus\{0\}\rightarrow U_N$ Ă© dada por:
	$$\phi_N^{-1}(u)=\frac{\left(2u^1,\dots,2u^n,|u|^2-1\right)}{|u|^2+1}$$
	
	\textbf{2. Suavidade das TransiĂ§Ă”es}
	
	A transiĂ§ĂŁo $\psi = \phi_S \circ \phi_N^{-1}$ Ă© definida no domĂ­nio $\phi_N(U_N \cap U_S) = \mathbb{R}\setminus\{0\}$. AlĂ©m disso,
	\begin{align*}
		\psi^i(u) = \frac{x^i}{1+x^{n+1}} = \frac{\frac{2u^i}{|u|^2+1}}{1 + \frac{|u|^2-1}{|u|^2+1}} = \frac{2u^i}{|u|^2+1+|u|^2-1} = \frac{2u^i}{2|u|^2} = \frac{u^i}{|u|^2}
	\end{align*}
	A expressĂŁo $\psi(u) = \frac{u}{|u|^2}$ Ă© suave em $\mathbb{R}^n \setminus \{0\}$, concluindo que o atlas $\{(U_N,\phi_N),(U_S,\phi_S)\}$ Ă© suave. Portanto, $\mathbb{S}^n$ Ă© uma $n$-variedade com essa estrutura.
	
	\textbf{(d)} NĂŁo Ă© possĂ­vel. Se existisse uma Ășnica carta $\phi: \mathbb{S}^n \to \mathbb{R}^n$, ela seria um homeomorfismo sobre um aberto $V \subseteq \mathbb{R}^n$. Pelo item (a), $\mathbb{S}^n$ Ă© compacto, logo $V$ deveria ser um subconjunto compacto e aberto de $\mathbb{R}^n$. Como $\mathbb{R}^n$ Ă© conexo, o Ășnico aberto e fechado (compacto) Ă© o conjunto vazio ou o prĂłprio $\mathbb{R}^n$ (que nĂŁo Ă© compacto). Como $\mathbb{S}^n \neq \emptyset$, chegamos a uma contradiĂ§ĂŁo.
}

	\[ y = \frac{x}{\sqrt{1-|x|^2}} \]

	\textbf{Suavidade de $f^{-1}$:} Analogamente a $f$, a aplicaĂ§ĂŁo $f^{-1}$ Ă© composta por funĂ§Ă”es suaves, pois o denominador $\sqrt{1+|y|^2}$ Ă© sempre maior ou igual a $1$, nunca se anulando para qualquer $y \in \mathbb{R}^n$. Logo, $f^{-1}$ Ă© suave.

	Consideremos a funĂ§ĂŁo $\psi : \mathbb{R} \to \mathbb{R}$ definida por:

	\end{equation*}

	\end{itemize}
